const [filename, setFilename] = useState("")
  const [accesstoken, setAccessToken] = useState("")
  const [username, setUsername] = useState("")
  const [files, setFiles] = useState([])

  const getFilename = (val) => {
    setFilename(val)
  }

  useEffect(() => {
    const getToken = async () => {

      //const username = await Auth.currentAuthenticatedUser()
      console.log(username.username)

      if (username) {
        setUsername(username.username)
      }

    }
    getToken()
  }, [])

  useEffect(() => {
    const getFiles = async () => {

      try {
        const user = await Auth.currentSession()
        const token = user.accessToken.jwtToken
        setAccessToken(token)
        const tokenValue = {
          'access_Token': token
        }
        let res = await fetch("https://pimvhp4mb5.execute-api.eu-north-1.amazonaws.com/dev/savedfiles",
          {
            method: 'POST',

            headers: {
              'Content-Type': 'application/json',
            },
            mode: 'cors',
            body: JSON.stringify(tokenValue)
          })
        if (res.ok) {
          const details = await res.json()
          //console.log(details)
          if (details.body) {
            const result = details.body
            setFiles(result)
          }
        }
        else {
          console.log(res)
        }
      } catch (err) {
        console.log(err)
      }
    }
    getFiles()
  }, [files])
